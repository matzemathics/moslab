428l4_msgtag_t l4_msgtag(long label, unsigned words, unsigned items,
429 unsigned flags) L4_NOTHROW
430{
431 return (l4_msgtag_t){ (l4_mword_t)((l4_umword_t)label << 16)
432 | (l4_mword_t)(words & 0x3f)
433 | (l4_mword_t)((items & 0x3f) << 6)
434 | (l4_mword_t)(flags & 0xf000)};
435}
