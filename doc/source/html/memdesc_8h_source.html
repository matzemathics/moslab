188L4_INLINE
189l4_kernel_info_mem_desc_t *
190l4_kernel_info_get_mem_descs(l4_kernel_info_t *kip) L4_NOTHROW
191{
192 return (l4_kernel_info_mem_desc_t *)(((l4_addr_t)kip)
193 + (kip->mem_info >> (sizeof(l4_umword_t) * 4)));
194}
195
196L4_INLINE
197unsigned
198l4_kernel_info_get_num_mem_descs(l4_kernel_info_t *kip) L4_NOTHROW
199{
200 return kip->mem_info & ((1UL << (sizeof(l4_umword_t)*4)) -1);
201}
202
203L4_INLINE
204void
205l4_kernel_info_set_mem_desc(l4_kernel_info_mem_desc_t *md,
206 l4_addr_t start,
207 l4_addr_t end,
208 unsigned type,
209 unsigned virt,
210 unsigned sub_type) L4_NOTHROW
211{
212 md->l = (start & ~0x3ffUL) | (type & 0x0f) | ((sub_type << 4) & 0x0f0)
213 | (virt ? 0x200 : 0x0);
214 md->h = end;
215}
