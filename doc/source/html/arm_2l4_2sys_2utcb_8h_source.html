81L4_INLINE l4_utcb_t *l4_utcb_direct(void) L4_NOTHROW
82{
83# if defined(__ARM_ARCH) && __ARM_ARCH >= 7
84 l4_utcb_t *utcb;
85 __asm__ ("mrc p15, 0, %0, c13, c0, 2" : "=r" (utcb)); // TPIDRURW
86#else
87 register l4_utcb_t *utcb __asm__ ("r0");
88 __asm__ ("mov lr, pc \n"
89 "mvn pc, #0xff \n" // write 0xffffff00 to pc
90 : "=r"(utcb) : : "lr");
91#endif
92 return utcb;
93}
