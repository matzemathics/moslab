129typedef struct l4virtio_config_hdr_t
130{
131 /* Virtio(0x00): device config */
132 l4_uint32_t magic;
133 l4_uint32_t version;
134 l4_uint32_t device;
135 l4_uint32_t vendor;
137 /* Virtio(0x10): device features */
138 l4_uint32_t dev_features;
139 l4_uint32_t dev_features_sel;
140 l4_uint32_t _res1[2];
141
142 /* Virtio(0x20): driver features */
143 l4_uint32_t driver_features;
144 l4_uint32_t driver_features_sel;
145
146 /* L4Virtio(0x28): L4 queue */
147 l4_uint32_t num_queues;
148 l4_uint32_t queues_offset;
150 /* Virtio(0x30): queue status */
151 l4_uint32_t queue_sel;
152 l4_uint32_t queue_num_max;
153 l4_uint32_t queue_num;
154 l4_uint32_t _res3[2];
155 l4_uint32_t queue_ready;
156 l4_uint32_t _res4[2];
157
158 /* Virtio(0x50): queue notify */
159 l4_uint32_t queue_notify;
160 l4_uint32_t _res5[3];
161
162 /* Virtio(0x60): interrupt handling */
163 l4_uint32_t irq_status;
164 l4_uint32_t irq_ack;
165 l4_uint32_t _res6[2];
166
167 /* Virtio(0x70): Device status register (read-only). The register must be
168 * written using l4virtio_set_status(). */
169 l4_uint32_t status;
170
171 /* L4Virtio(0x74): W: Event index to be used for config notifications (device to driver) */
172 l4_uint32_t cfg_driver_notify_index;
173 /* L4Virtio(0x78): R: Event index to be used for config notifications (driver to device) */
174 l4_uint32_t cfg_device_notify_index;
175
176 /* L4Virtio(0x7c) L4 specific command register polled by the driver iff supported */
177 l4_uint32_t cmd;
178
179 /* Virtio(0x80): queue descriptors */
180 l4_uint64_t queue_desc;
181 l4_uint32_t _res8[2];
182 l4_uint64_t queue_avail;
183 l4_uint32_t _res9[2];
184 l4_uint64_t queue_used;
185
186 l4_uint32_t _res10[1];
187
188 /* Virtio(0xac): shared memory region */
189 l4_uint32_t shm_sel;
190 l4_uint64_t shm_len;
191 l4_uint64_t shm_base;
192
193 /* L4Virtio(0xc0): use the unused space here for device and driver feature bitmaps */
194 l4_uint32_t dev_features_map[6];
195 l4_uint32_t _res11[2];
196 l4_uint32_t driver_features_map[6];
197 l4_uint32_t _res12[1];
198
199 /* Virtio(0xfc): config generation */
200 l4_uint32_t generation;
201} l4virtio_config_hdr_t;

268L4_INLINE void
269l4virtio_set_feature(l4_uint32_t *feature_map, unsigned feat)
270{
271 unsigned idx = feat / 32;
272
273 if (idx < 8)
274 feature_map[idx] |= 1UL << (feat % 32);
275}
276
280L4_INLINE void
281l4virtio_clear_feature(l4_uint32_t *feature_map, unsigned feat)
282{
283 unsigned idx = feat / 32;
284
285 if (idx < 8)
286 feature_map[idx] &= ~(1UL << (feat % 32));
287}
288
292L4_INLINE unsigned
293l4virtio_get_feature(l4_uint32_t *feature_map, unsigned feat)
294{
295 unsigned idx = feat / 32;
296
297 if (idx >= 8)
298 return 0;
299
300 return feature_map[idx] & (1UL << (feat % 32));
301}
