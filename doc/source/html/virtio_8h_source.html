254L4_INLINE void
255l4virtio_set_feature(l4_uint32_t *feature_map, unsigned feat)
256{
257 unsigned idx = feat / 32;
258
259 if (idx < 8)
260 feature_map[idx] |= 1UL << (feat % 32);
261}
262
266L4_INLINE void
267l4virtio_clear_feature(l4_uint32_t *feature_map, unsigned feat)
268{
269 unsigned idx = feat / 32;
270
271 if (idx < 8)
272 feature_map[idx] &= ~(1UL << (feat % 32));
273}
274
278L4_INLINE unsigned
279l4virtio_get_feature(l4_uint32_t *feature_map, unsigned feat)
280{
281 unsigned idx = feat / 32;
282
283 if (idx >= 8)
284 return 0;
285
286 return feature_map[idx] & (1UL << (feat % 32));
287}
