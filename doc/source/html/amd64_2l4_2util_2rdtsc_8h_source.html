166L4_INLINE l4_cpu_time_t
167l4_rdtsc (void)
168{
169 l4_umword_t lo, hi;
170
171 __asm__ __volatile__ ("rdtsc" : "=a"(lo), "=d"(hi));
172
173 return ((l4_cpu_time_t)hi << 32) | lo;
174}

206L4_INLINE l4_uint64_t
207l4_tsc_to_ns (l4_cpu_time_t tsc)
208{
209 l4_uint64_t ns, dummy;
210 __asm__
211 (" \n\t"
212 "mulq %3 \n\t"
213 "shrd $27, %%rdx, %%rax \n\t"
214 :"=a" (ns), "=d"(dummy)
215 :"a" (tsc), "r" ((l4_uint64_t)l4_scaler_tsc_to_ns)
216 );
217 return ns;
218}

234L4_INLINE void
235l4_tsc_to_s_and_ns (l4_cpu_time_t tsc, l4_uint32_t *s, l4_uint32_t *ns)
236{
237 __asm__
238 (" \n\t"
239 "mulq %3 \n\t"
240 "shrd $27, %%rdx, %%rax \n\t"
241 "xorq %%rdx, %%rdx \n\t"
242 "divq %4 \n\t"
243 :"=a" (*s), "=&d" (*ns)
244 : "a" (tsc), "r" ((l4_uint64_t)l4_scaler_tsc_to_ns),
245 "rm"(1000000000ULL)
246 );
247}

263L4_INLINE void
264l4_busy_wait_ns (l4_uint64_t ns)
265{
266 l4_cpu_time_t stop = l4_rdtsc();
267 stop += l4_ns_to_tsc(ns);
268
269 while (l4_rdtsc() < stop)
