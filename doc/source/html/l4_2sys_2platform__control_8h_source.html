201 l4_umword_t extras,
202 l4_utcb_t *utcb) L4_NOTHROW
203{
204 l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
205 v->mr[0] = L4_PLATFORM_CTL_SYS_SUSPEND_OP;
206 v->mr[1] = extras;
207 return l4_ipc_call(pfc, utcb, l4_msgtag(L4_PROTO_PLATFORM_CTL, 2, 0, 0),
208 L4_IPC_NEVER);
