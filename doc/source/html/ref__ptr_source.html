28template< typename T >
29struct Default_ref_counter
30{
31 void h_drop_ref(T *p) noexcept
32 {
33 if (p->remove_ref() == 0)
34 delete p;
35 }
36
37 void h_take_ref(T *p) noexcept
38 {
39 p->add_ref();
40 }
41};
